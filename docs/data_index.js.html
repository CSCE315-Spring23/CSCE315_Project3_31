import backendOrigin from "../config/origins";
import uuid from "react-uuid";

	/**
	 * Login modal popup that shows when user joins and when he/she wishes to swich views.
	 * @param model {string} 
	 * @param params {json} 
	 */
	static async _getModel(model, params) {
		let time = Date.now();
		const response = await fetch(`${backendOrigin}/${model}`, params)
			.then((res) => res.json())
			.then((json) => {
				return json;
			})
			.catch((err) => {
				console.log(err);
			});
		return response;
	}

	static async makeOrder(
		cost_total,
		timestamp,
		customer_id,
		staff_id,
		menu_items
	) {
		const response = await fetch(`${backendOrigin}/orders/makeorder`, {
			method: "POST",
			headers: {
				"Content-Type": "application/json",
			},
			body: JSON.stringify({
				cost_total: cost_total,
				timestamp: timestamp,
				customer_id: customer_id,
				staff_id: staff_id,
				menu_items: menu_items,
			}),
		})

export const handleUID = () => {
	let uid = localStorage.getItem("uid");
	if (!uid) {
		uid = uuid();
		localStorage.setItem("uid", uid);
	}
};

export const getUID = () => {
	handleUID();
	let uid = localStorage.getItem("uid");
	return uid;
};
